### 1. EvaluationService 확장

```python
class EvaluationService:
    
    def __init__(self, db: Session):
        self.db = db
        self.graph_service = CourseGraphService()  # Neo4j 연동
        self._similarity_cache = {}  # 유사도 캐시
        self._similarity_threshold = 0.7  # 유사과목 인정 최소 유사도
    
    def _get_similarity_from_graph(
        self, 
        source_course_code: str, 
        target_course_code: str
    ) -> float:
        """
        Neo4j에서 두 과목 간 유사도 조회
        
        Returns:
            유사도 (0.0 ~ 1.0), 관계 없으면 0.0
        """
        # 캐시 확인
        cache_key = tuple(sorted([source_course_code, target_course_code]))
        if cache_key in self._similarity_cache:
            return self._similarity_cache[cache_key]
        
        # Neo4j 조회
        similarity = self.graph_service.get_similarity_between(
            source_course_code, target_course_code
        )
        
        self._similarity_cache[cache_key] = similarity
        return similarity
    
    def _find_best_similar_course(
        self,
        target_course_code: str,
        target_course_name: str,
        student_completed_courses: Dict
    ) -> Tuple[bool, float, Optional[Dict]]:
        """
        학생이 이수한 과목 중 타겟 과목과 가장 유사한 과목 찾기
        
        Returns:
            (인정여부, 최고유사도, 매칭된과목)
        """
        completed_codes = student_completed_courses["codes"]
        completed_details = student_completed_courses["details"]
        
        # 1. 먼저 동일 과목 체크
        if target_course_code in completed_codes:
            for detail in completed_details:
                if detail["course_code"] == target_course_code:
                    return (True, 1.0, detail)  # 동일과목 = 유사도 1.0
        
        # 2. Neo4j에서 유사 과목 조회
        best_similarity = 0.0
        best_match = None
        
        for detail in completed_details:
            sim = self._get_similarity_from_graph(
                detail["course_code"], 
                target_course_code
            )
            if sim > best_similarity:
                best_similarity = sim
                best_match = detail
        
        # 3. 임계값 이상이면 인정
        is_accepted = best_similarity >= self._similarity_threshold
        return (is_accepted, best_similarity, best_match if is_accepted else None)
```

### 2. CourseGraphService 확장

```python
# graph_service.py에 추가

@staticmethod
def get_similarity_between(code1: str, code2: str) -> float:
    """
    두 과목 간 유사도 조회
    
    Args:
        code1: 첫 번째 과목 학수번호
        code2: 두 번째 과목 학수번호
        
    Returns:
        유사도 (0.0 ~ 1.0), 관계 없으면 0.0
    """
    with get_session() as session:
        result = session.run(
            """
            MATCH (c1:Course {code: $code1})-[r:SIMILAR_TO]-(c2:Course {code: $code2})
            RETURN r.similarity as similarity
            LIMIT 1
            """,
            code1=code1,
            code2=code2
        )
        record = result.single()
        return float(record['similarity']) if record else 0.0

@staticmethod
def get_similar_courses_for_list(
    target_codes: List[str], 
    min_similarity: float = 0.7
) -> Dict[str, List[Dict]]:
    """
    여러 과목에 대해 유사 과목 일괄 조회 (배치 최적화)
    
    Returns:
        {타겟학수번호: [{code, name, similarity}, ...]}
    """
    with get_session() as session:
        result = session.run(
            """
            MATCH (c1:Course)-[r:SIMILAR_TO]-(c2:Course)
            WHERE c1.code IN $codes AND r.similarity >= $min_sim
            RETURN c1.code as target_code,
                   c2.code as similar_code,
                   c2.name as similar_name,
                   r.similarity as similarity
            ORDER BY c1.code, r.similarity DESC
            """,
            codes=target_codes,
            min_sim=min_similarity
        )
        
        grouped = {}
        for record in result:
            target = record['target_code']
            if target not in grouped:
                grouped[target] = []
            grouped[target].append({
                'code': record['similar_code'],
                'name': record['similar_name'],
                'similarity': record['similarity']
            })
        return grouped
```
